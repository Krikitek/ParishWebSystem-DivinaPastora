CREATE TABLE FirstCommunionServiceTable (
    firstCommunionServiceID INT IDENTITY(1,1) PRIMARY KEY,

    lastName NVARCHAR(50) NOT NULL,
    firstName NVARCHAR(50) NOT NULL,
    middleName NVARCHAR(50),
    gender NVARCHAR(10),

    province NVARCHAR(100),
    city NVARCHAR(100),

    dateOfBirth DATE,
    baptizedAt NVARCHAR(255),
    age INT,

    fatherLname NVARCHAR(50),
    fatherFname NVARCHAR(50),
    fatherMname NVARCHAR(50),

    motherLname NVARCHAR(50),
    motherFname NVARCHAR(50),
    motherMname NVARCHAR(50)
);
